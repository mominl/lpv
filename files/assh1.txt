#include <iostream>
#include <omp.h>

using namespace std;

#define MAX 100

class Graph {
    int V;
    int adj[MAX][MAX];

public:
    Graph(int vertices) {
        V = vertices;

        // Initialize adjacency matrix
        for (int i = 0; i < MAX; i++) {
            for (int j = 0; j < MAX; j++) {
                adj[i][j] = 0;
            }
        }
    }

    // Add edge
    void addEdge(int u, int v) {
        adj[u][v] = 1;
        adj[v][u] = 1;
    }

    // Parallel BFS
    void parallelBFS(int start) {

        bool visited[MAX] = {false};
        int frontier[MAX], next_frontier[MAX];

        int frontier_size = 0;
        int next_size = 0;

        frontier[frontier_size++] = start;
        visited[start] = true;

        cout << "Parallel BFS: ";

        while (frontier_size > 0) {

            next_size = 0;

#pragma omp parallel for
            for (int i = 0; i < frontier_size; i++) {

                int node = frontier[i];

#pragma omp critical
                cout << node << " ";

                for (int j = 0; j < V; j++) {

                    if (adj[node][j] == 1 && !visited[j]) {

#pragma omp critical
                        {
                            if (!visited[j]) {
                                visited[j] = true;
                                next_frontier[next_size++] = j;
                            }
                        }
                    }
                }
            }

            // Copy next frontier
            frontier_size = next_size;

            for (int i = 0; i < next_size; i++) {
                frontier[i] = next_frontier[i];
            }
        }

        cout << endl;
    }

    // Parallel DFS Utility
    void parallelDFSUtil(int node, bool visited[]) {

        visited[node] = true;

#pragma omp critical
        cout << node << " ";

        for (int i = 0; i < V; i++) {

            if (adj[node][i] == 1 && !visited[i]) {

#pragma omp task
                parallelDFSUtil(i, visited);
            }
        }
    }

    // Parallel DFS
    void parallelDFS(int start) {

        bool visited[MAX] = {false};

        cout << "Parallel DFS: ";

#pragma omp parallel
        {
#pragma omp single
            {
                parallelDFSUtil(start, visited);
            }
        }

        cout << endl;
    }
};

int main() {

    int V, E;

    cout << "Enter number of vertices: ";
    cin >> V;

    Graph g(V);

    cout << "Enter number of edges: ";
    cin >> E;

    cout << "Enter edges (u v):\n";

    for (int i = 0; i < E; i++) {

        int u, v;
        cin >> u >> v;

        if (u >= V || v >= V) {
            cout << "Invalid edge!\n";
            i--;
            continue;
        }

        g.addEdge(u, v);
    }

    int start;

    cout << "Enter starting node: ";
    cin >> start;

    g.parallelBFS(start);
    g.parallelDFS(start);

    return 0;
}
